## Arrow Function
Sebelum ES6
```js
hello = function() {
  return "Hello World!"
}
```
Setelah ES6
```js
// jika function hanya return satu value
hello = () => "Hello World!"

hello = () => {
  return "Hello World!"
}
```  

Sebelum ES6
```js
let myFunction = function(a, b) {
	return a * b
}
```  
Setelah ES6
```js
// jika function hanya return satu value
let myFunction = (a, b) => a * b;


let myFunction =  (a, b) => {
  return a * b
}
```
